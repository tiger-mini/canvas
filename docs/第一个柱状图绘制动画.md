### 第一个Canvas的柱状条绘制动画


#### 



```html

<html>
    <head>
       <title>demo</title>
       <meta name="description" content="模板">
       <meta name="keywords" content="HTML,CSS,XML,JavaScript">
       <meta name="author" content="yangb03">
       <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1, user-scalable=no">
        <style type="text/css">
            canvas{
              box-shadow: 0 0 10px #e5e5e5;
            }
        </style>
    </head>
    <body>
       <canvas id='canvas' width="600" height='400'></canvas>
       <script type="text/javascript">
          window.onload = function() {

            var tween = {
              linear:function(t,b,c,d){
                return c*t/d + b;
              },
              easeIn:function(t,b,c,d){
                return c * ( t /= d ) * t + b;
              },
              easeOut: function (t, b, c, d) {
                if ((t /= d) < (1 / 2.75)) {
                  return c * (7.5625 * t * t) + b;
                } else if (t < (2 / 2.75)) {
                  return c * (7.5625 * (t -= (1.5 / 2.75)) * t + .75) + b;
                } else if (t < (2.5 / 2.75)) {
                  return c * (7.5625 * (t -= (2.25 / 2.75)) * t + .9375) + b;
                } else {
                  return c * (7.5625 * (t -= (2.625 / 2.75)) * t + .984375) + b;
                }
              },
              strongEaseIn:function(t,b,c,d){
                return c * ( t /= d ) * t * t * t * t + b;
              },
              strongEaseOut:function(t,b,c,d){
                return c * ( ( t = t / d -1 ) * t * t * t * t +1 ) + b;
              },
              sineaseIn:function(t,b,c,d){
                return c * ( t /= d ) * t * t + b;
              },
              sineaseOut:function(t,b,c,d){
                return c * ( ( t = t / d -1 ) * t * t *t +1 ) + b;
              },
            };

            const canvas = document.getElementById('canvas');
            const ctx = canvas.getContext('2d');

            

            let height = 0;
            let startHeight = 0;
            let lastHeight = 100;
            let startDate = Date.now();
            let duration = 1000;

            function drawRect() {
              ctx.clearRect(0, 0, canvas.width, canvas.height);
              ctx.beginPath();
              ctx.rect(200, 400, 80 , -height);
              ctx.fill();
              ctx.stroke();
            }
            function animateRect() {
              console.log('cc');
              const t = Date.now();
              if(t >= startDate + duration) {
                  return ;
              }
              var pos = tween.strongEaseOut(t - startDate, startHeight, 100, duration);
              height = pos;
              drawRect();
              window.requestAnimationFrame(animateRect)
            }
            animateRect();
          }
       </script>
    </body>
</html>


```
